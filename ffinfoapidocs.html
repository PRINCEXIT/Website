Python Example

import requests
from rich.console import Console
from rich.panel import Panel
from rich.table import Table
from rich.tree import Tree
from rich.text import Text
from rich.layout import Layout
from rich.style import Style
from rich import box
from datetime import datetime
from rich.prompt import Prompt
from rich.progress import Progress, SpinnerColumn, TextColumn

console = Console()

def get_player_stats(uid, region="sg", api_key="PRINCE-PERM-KEY"):
    url = f"https://freefireinfoapiv2lk-team.vercel.app/api/playerstats"
    params = {
        "uid": uid,
        "region": region.lower(),
        "api_key": api_key
    }
    
    try:
        response = requests.get(url, params=params)
        response.raise_for_status()
        return response.json()
    except requests.exceptions.RequestException as e:
        console.print(f"[red]Error fetching data: {str(e)}[/red]")
        return None

def format_timestamp(timestamp):
    try:
        return datetime.fromtimestamp(int(timestamp)).strftime('%Y-%m-%d %H:%M:%S')
    except:
        return "N/A"

def create_stats_table(title, data, columns):
    table = Table(title=title, box=box.ROUNDED, style="cyan", title_style="bold magenta")
    for col in columns:
        table.add_column(col[0], style=col[1])
    for row in data:
        table.add_row(*[str(item) for item in row])
    return table

def display_player_info(data):
    layout = Layout()
    layout.split_column(
        Layout(name="header"),
        Layout(name="main", ratio=3),
        Layout(name="footer")
    )
    
    # Header with Player Basic Info
    account_info = data.get("AccountInfo", {})
    header_panel = Panel(
        f"""[bold cyan]Player:[/bold cyan] {account_info.get('AccountName')}  |  [bold cyan]UID:[/bold cyan] {data.get('GuildInfo', {}).get('GuildOwner')}
[bold cyan]Level:[/bold cyan] {account_info.get('AccountLevel')}  |  [bold cyan]Region:[/bold cyan] {account_info.get('AccountRegion')}  |  [bold cyan]Release Version:[/bold cyan] {account_info.get('ReleaseVersion')}""",
        style="bold white on blue",
        title="üéÆ Free Fire Player Profile",
        box=box.DOUBLE
    )
    layout["header"].update(header_panel)

    main_tree = Tree("[bold green]Detailed Information")

    account_panel = Panel(
        f"""[bold yellow]Experience:[/bold yellow] {account_info.get('AccountEXP'):,}
[bold yellow]Account Created:[/bold yellow] {format_timestamp(account_info.get('AccountCreateTime'))}
[bold yellow]Last Login:[/bold yellow] {format_timestamp(account_info.get('AccountLastLogin'))}
[bold yellow]Likes:[/bold yellow] {account_info.get('AccountLikes'):,}
[bold yellow]Battle Royale Rank:[/bold yellow] {account_info.get('BrMaxRank')} (Points: {account_info.get('BrRankPoint'):,})
[bold yellow]Clash Squad Rank:[/bold yellow] {account_info.get('CsMaxRank')} (Points: {account_info.get('CsRankPoint'):,})
[bold yellow]Elite Pass Status:[/bold yellow] {'‚úÖ Active' if account_info.get('hasElitePass') else '‚ùå Inactive'}""",
        title="üìä Account Statistics",
        box=box.ROUNDED
    )
    main_tree.add(account_panel)

    guild_info = data.get("GuildInfo", {})
    guild_panel = Panel(
        f"""[bold green]Guild Name:[/bold green] {guild_info.get('GuildName')}
[bold green]Guild ID:[/bold green] {guild_info.get('GuildID')}
[bold green]Guild Level:[/bold green] {guild_info.get('GuildLevel')}
[bold green]Members:[/bold green] {guild_info.get('GuildMember')}/{guild_info.get('GuildCapacity')}
[bold green]Guild Owner:[/bold green] {guild_info.get('GuildOwner')}""",
        title="üè∞ Guild Information",
        box=box.ROUNDED
    )
    main_tree.add(guild_panel)

    layout["main"].update(main_tree)
    footer_text = Text("Data retrieved at: " + datetime.now().strftime('%Y-%m-%d %H:%M:%S'), style="italic cyan")
    layout["footer"].update(Panel(footer_text, box=box.ROUNDED))
    console.print(layout)

def main():
    while True:
        uid = Prompt.ask("[bold cyan]Enter player UID[/bold cyan]")
        region = Prompt.ask(
            "[bold cyan]Enter region code[/bold cyan]",
            choices=["sg", "in", "br", "id", "th", "vn", "ru", "me", "pk", "bd"],
            default="sg"
        )

        with Progress(
            SpinnerColumn(),
            TextColumn("[progress.description]{task.description}"),
            transient=True,
        ) as progress:
            progress.add_task(description="Fetching player data...", total=None)
            player_data = get_player_stats(uid, region)

        if player_data:
            display_player_info(player_data)
            if not Prompt.ask("\n[yellow]Check another player?[/yellow]", choices=["y", "n"], default="y") == "y":
                break
        else:
            console.print("[red]Failed to fetch player data[/red]")
            if not Prompt.ask("\n[yellow]Try again?[/yellow]", choices=["y", "n"], default="y") == "y":
                break

if __name__ == "__main__":
    try:
        main()
    except KeyboardInterrupt:
        console.print("\n[yellow]Exiting...[/yellow]")
    except Exception as e:
        console.print(f"\n[red]Error: {str(e)}[/red]")
